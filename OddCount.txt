class OddCount
{  
    public void count(int low, int high) 
    {
        int count=0;
        for(int i=low;i<=high;i++)
            if(i%2==1) 
                count++;
        
        System.out.println(count);
    }
    
    public static void main(String args[])
    {
        int low=12,high=15;
        OddCount c= new OddCount();
        c.count(low, high);
    }
}
